## Anatomy

```tsx
import { Menu, MenuContent, MenuItem, MenuTrigger } from "@/components/ui/menu";

<Menu>
  <MenuTrigger render={<Button variant="secondary">Open</Button>} />
  <MenuContent>
    <MenuItem>Profile</MenuItem>
    <MenuItem>Settings</MenuItem>
  </MenuContent>
</Menu>;
```
