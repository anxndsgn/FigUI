## Anatomy

```tsx
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from '@/components/ui/select';

<Select items={items}>
  <SelectTrigger iconLead={<RulerIcon />}>
    <SelectValue />
  </SelectTrigger>
  <SelectContent>
    <SelectItem value='arial'>Arial</SelectItem>
  </SelectContent>
</Select>
```

## Props

### SelectTrigger

| Prop | Type | Description |
| ---- | ---- | ----------- |
| `iconLead` | `React.ReactNode` | Optional leading icon or label shown before the value |
